void theShieldIsUp (int param) {
  raise(SIGABRT);
}

void ackbar (int param) {
  std::cout << "It's A Trap!" << std::endl;
  exit(1);
}

int main () {
  signal(SIGTERM, theShieldIsUp);
  signal(SIGABRT, ackbar);
  while(true)
    sleep(1);
  return 0;
}
